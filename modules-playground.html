Named Exports & Imports

Module File (math.js)

// Named exports export const PI = 3.14159; export function add(a, b) { return a + b; } export function multiply(a, b) { return a * b; } export class Calculator { constructor() { this.result = 0; } add(value) { this.result += value; return this; } }

Import File (main.js)

// Import specific named exports import { PI, add, multiply } from './math.js'; // Import with alias import { add as sum, multiply as product } from './math.js'; // Import all named exports import * as math from './math.js'; // Usage console.log(PI); // 3.14159 console.log(add(5, 3)); // 8 console.log(math.multiply(4, 6)); // 24

Default Exports & Imports

Module File (utils.js)

// Default export - class export default class Logger { constructor(name) { this.name = name; } log(message) { console.log(`[${this.name}] ${message}`); } } // You can also have named exports alongside default export const VERSION = '1.0.0';

Import File (main.js)

// Import default export import Logger from './utils.js'; // Import default with alias import MyLogger from './utils.js'; // Import both default and named import Logger, { VERSION } from './utils.js'; // Usage const logger = new Logger('MyApp'); logger.log('Hello World'); // [MyApp] Hello World console.log(VERSION); // 1.0.0

Mixed Export Types

Module File (api.js)

// Default export export default class APIClient { constructor(baseURL) { this.baseURL = baseURL; } } // Named exports export const GET = 'GET'; export const POST = 'POST'; export const PUT = 'PUT'; export function request(method, url, data) { return fetch(url, { method, headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }); }

Import File (main.js)

// Import default and named exports import APIClient, { GET, POST, request } from './api.js'; // Or import all import * as API from './api.js'; // Usage const client = new APIClient('https://api.example.com'); console.log(GET, POST); // GET POST // Using named import request(POST, '/users', { name: 'John' }); // Using namespace import API.request(API.GET, '/users');

Dynamic Imports

Module File (heavy.js)

// Heavy module that loads on demand export function heavyFunction() { // Simulate heavy computation let result = 0; for (let i = 0; i < 1000000; i++) { result += Math.random(); } return result; } export const config = { debug: true, version: '2.0.0' };

Import File (main.js)

// Dynamic import - loads module when needed async function loadHeavyModule() { try { const module = await import('./heavy.js'); console.log(module.heavyFunction()); console.log(module.config); } catch (error) { console.error('Failed to load module:', error); } } // Conditional import async function loadModuleBasedOnCondition(useHeavy) { if (useHeavy) { const { heavyFunction } = await import('./heavy.js'); return heavyFunction(); } return 'Light version'; }
